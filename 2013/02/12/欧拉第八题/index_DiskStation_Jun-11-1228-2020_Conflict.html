def maxProduct(number):
    num=list(str(number))
	maxProduct=1
	for i in range(len(num)):
		tmp=num[i:i+5]
		if "0" in tmp:
            continue
		else:
            product=1
            for ele in tmp:
                product=product*int(ele) 
             	if product>maxProduct: 
                 	maxProduct=product 
                
	return maxProduct
